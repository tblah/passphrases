# titles
set title "Passphrase Complexity By Length and Character Space
set xlabel "Character Space"
set ylabel "Passphrase Length"
set zlabel "Complexity" rotate by 90

# look pretty
set isosamples 70
set samples 60

# ranges
set xrange[1:10]
set yrange[1:10]
set zrange[1:1000]

# set viewing angle
set view 38, 350

# plot it
f(x,y) = (((x**y)-1)/(y-1))-1
splot f(x,y)
